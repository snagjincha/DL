1. RNNCell

x = torch.tensor(df_train.x.map({'A':0,'b':1,'c':2,'d':3}))
y = torch.tensor(df_train.y.map({'A':0,'b':1,'c':2,'d':3}))
X = torch.nn.functional.one_hot(x).float()
y = torch.nn.functional.one_hot(y).float()
rnncell = torch.nn.RNNCell(4,3)
linr = torch.nn.Linear(3,4)
loss_fn = torch.nn.CrossEntropyLoss()
optimizr = torch.optim.Adam(list(rnncell.parameters()) + list(linr.parameters()),lr=0.1)
L = len(X)
for l in range(200):
    loss = 0
    ht = torch.zeros(3)
    for t in range(L):
        Xt,yt = X[t],y[t]
        ht = rnncell(Xt,ht)
        netout = linr(ht)
        loss = loss + loss_fn(netout,yt)
    loss = loss/L
    loss.backward()
    optimizr.step()
    optimizr.zero_grad()
h = torch.zeros(L,3)
water = torch.zeros(3)
h[0] = rnncell(X[0],water)
for t in range(1,L):

5. torch.nn.Module + rNNCell

x = torch.tensor(df_train.x.map({'A':0,'b':1,'c':2,'d':3}))

X = torch.nn.functional.one_hot(x).float()
y = torch.nn.functional.one_hot(y).float()
class rNNCell(torch.nn.Module):
    def __init__(self):
        super().__init__()
        self.i2h = torch.nn.Linear(4,2) 
        self.h2h = torch.nn.Linear(2,2) 
        self.tanh = torch.nn.Tanh()
    def forward(self,Xt,ht):
        ht = self.tanh(self.i2h(Xt)+self.h2h(ht))
        return ht
